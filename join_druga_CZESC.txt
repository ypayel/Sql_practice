--Zadanie 11

SELECT 
    P.ProductID,
    P.ProductName,
    S.SupplierID,
    S.CompanyName AS SupplierName
FROM dbo.Products P
FULL OUTER JOIN dbo.Suppliers S
      ON P.SupplierID = S.SupplierID
ORDER BY 
    COALESCE(P.ProductID, -1);

--Zadanie 14

SELECT
    P.ProductID,
    P.ProductName,
    C.CategoryID,
    C.CategoryName,
    S.SupplierID,
    S.CompanyName AS SupplierName
FROM 
    dbo.Products P
LEFT JOIN 
    dbo.Categories C ON P.CategoryID = C.CategoryID
LEFT JOIN 
    dbo.Suppliers S ON P.SupplierID = S.SupplierID
ORDER BY 
    P.ProductID;

--Zadanie 15

SELECT
    E.EmployeeID,
    E.FirstName,
    E.LastName,
    O.OrderID,
    O.OrderDate
FROM 
    dbo.Employees E
FULL OUTER JOIN 
    dbo.Orders O ON E.EmployeeID = O.EmployeeID
ORDER BY 
    EmployeeID;
